\subsection*{Question 2: Cas gÃĐnÃĐral : le prÃĐdicteur de Bayes de la perte absolue est la mÃĐdiane}

\paragraph{ConvexitÃĐ et minimisation.} Comme $p$ est continue, $F$ est dÃĐrivable

\]
La valeur $z$ telle que $F(z)=\frac12$ est prÃĐcisÃĐment une \textbf{mÃĐdiane} de la
loi conditionnelle. D'oÃđ
\[

\]
(Si plusieurs $z$ vÃĐrifient $F(z)=\frac12$, ils forment un intervalle sur lequel

\section{EspÃĐrance du risque empirique pour l'OLS}
% =====================================================================

On se place dans le modÃĻle linÃĐaire Ã  design fixe :

\[
    \E\big[R_n(\hat\theta)\big]=\frac{n-d}{n}\,\sigma^2,
    \qquad\text{oÃđ } R_n(\theta)=\tfrac1n\lVert y-X\theta\rVert^2 .

\emph{Remarque.} Les questions 2 Ã  6 calculent l'espÃĐrance du risque
\emph{empirique} (in-sample) $\E[R_n(\hat\theta)]=\frac{n-d}{n}\sigma^2$. Le

\subsection*{Question 1: Comparaison au risque de Bayes}

Dans ce modÃĻle, le meilleur prÃĐdicteur possible est $\theta^{*}$, et son risque

\[
    \E\big[R_n(\hat\theta)\big]=\frac{n-d}{n}\,\sigma^2

\textbf{Le risque empirique espÃĐrÃĐ est plus petit que le risque de Bayes.}
InterprÃĐtation : c'est l'\emph{optimisme} de l'erreur d'apprentissage. L'OLS

\begin{itemize}[nosep]
  \item Quand $d\ll n$ : $\E[R_n(\hat\theta)]\to\sigma^2$, l'erreur d'apprentissage
        est une bonne estimation de l'erreur rÃĐelle.
  \item Quand $d\to n$ : $\E[R_n(\hat\theta)]\to 0$, le modÃĻle interpole les
        donnÃĐes (sur-apprentissage) : l'erreur d'entraÃŪnement devient nulle bien

\subsection*{Question 2: RÃĐÃĐcriture avec la matrice de projection}

On note $P=X(X^TX)^{-1}X^T$ la matrice de projection orthogonale sur $\mathrm{Im}(X)$.
Les prÃĐdictions sont $X\hat\theta=X(X^TX)^{-1}X^Ty=Py$, donc le rÃĐsidu vaut

\[

\]
Or $(I_n-P)X=X-X(X^TX)^{-1}X^TX=X-X=0$, donc le premier terme est nul. Ainsi
$y-X\hat\theta=(I_n-P)\varepsilon$ et
\[
    R_n(\hat\theta)=\frac1n\lVert(I_n-P)\varepsilon\rVert^2
    \;\Longrightarrow\;
    \E\big[R_n(\hat\theta)\big]
       =\E_\varepsilon\!\left[\frac1n\big\lVert\big(I_n-X(X^TX)^{-1}X^T\big)\varepsilon\big\rVert^2\right].
\]

\subsection*{Question 3: $\sum_{i,j}A_{ij}^2=\Tr(A^TA)$}

Soit $A\in\R^{n\times n}$. Le coefficient diagonal $j$ de $A^TA$ est
$(A^TA)_{jj}=\sum_{i=1}^{n}(A^T)_{ji}A_{ij}=\sum_{i=1}^{n}A_{ij}^2$. En sommant sur $j$ :
\[
    \Tr(A^TA)=\sum_{j=1}^{n}(A^TA)_{jj}=\sum_{j=1}^{n}\sum_{i=1}^{n}A_{ij}^2
            =\sum_{(i,j)\in[1,n]^2}A_{ij}^2 .
\]

\subsection*{Question 4: $\E_\varepsilon[\frac1n\lVert A\varepsilon\rVert^2]=\frac{\sigma^2}{n}\Tr(A^TA)$}

\[

\]
En combinant : $A^TA=A\,A=A^2=A$. \hfill$\square$

\subsection*{Question 6: Conclusion}

\[
    \E\big[R_n(\hat\theta)\big]
